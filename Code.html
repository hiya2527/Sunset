                 $("#Soafbtn").click(
        function() {
            let num1 = parseInt(document.getElementById("InputSoaf1").value);
            let num2 = parseInt(document.getElementById("InputSoaf2").value);
            let num3 = parseInt(document.getElementById("InputSoaf3").value);
            let num4 = parseInt(document.getElementById("InputSoaf4").value);
            let num5 = parseInt(document.getElementById("InputSoaf5").value);

            let soafArr = [num1,num2,num3, num4, num5];
            let k = parseInt(document.getElementById("InputSoaf").value);
            for (let loop = 0; loop < soafArr.length; loop++) {
                let checkNum = k - soafArr[loop];
                for (let kLoop = loop + 1; kLoop < soafArr.length; kLoop++) {
                    if (checkNum == soafArr[kLoop]) {
                        document.getElementById("OutputSoaf").innerHTML += `Sum Found. ${soafArr[loop]} + ${soafArr[kLoop]} = ${k}.   `
                    }

                }
            }
        }
